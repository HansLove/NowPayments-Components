# NOWPayments Components

A comprehensive React component library for cryptocurrency payments powered by the NOWPayments API.

## Overview

## Quick Start

```bash
npm install @taloon/nowpayments-components
```

```tsx
import { DepositModal } from '@taloon/nowpayments-components'
import '@taloon/nowpayments-components/styles'

function App() {
  const [isOpen, setIsOpen] = useState(false)

  const handleSubmit = async (formData) => {
    // Your backend integration
    return await fetch('/api/deposits', {
      method: 'POST',
      body: JSON.stringify(formData)
    })
  }

  return (
    <DepositModal
      isOpen={isOpen}
      onClose={() => setIsOpen(false)}
      onSubmit={handleSubmit}
      customerEmail="user@example.com"
    />
  )
}
```
